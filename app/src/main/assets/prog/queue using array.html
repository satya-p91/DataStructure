Queue using Array

#include≺stdio.h≻
#include≺process.h≻
#define MAX 50
int queue_array[MAX];
int rear = - 1;
int front = - 1;

int insert(){
 int add_item;
 if (rear == MAX - 1)
  printf("Queue Overflow \n");
 else{
  if (front == - 1)
   front = 0;
   printf("Inset the element in queue : ");
  scanf("%d", &add_item);
  rear = rear + 1;
  queue_array[rear] = add_item;
 }
 return 0;
}

int delete(){
 if (front == - 1 || front > rear){
  printf("Queue Underflow \n");
  return 0;
 }
 else{
  printf("%d is deleted\n", queue_array[front]);
  front = front + 1;
 }
 return 0;
}

int display(){
 int i;
 if (front == - 1)
  printf("Queue is empty \n");
 else{
  printf("Queue is : \n");
  for (i = front; i <= rear; i++)
   printf("%d ", queue_array[i]);
  printf("\n");
 }
 return 0;
}

int main(){
 int choice;
  while (1){
  printf("1.Insert 2.delete 3.display 4.exit\n");
  printf("Enter your choice : ");
  scanf("%d", &choice);
  switch (choice){
   case 1: insert();
    break;
   case 2: delete();
    break;
   case 3: display();
    break;
    case 4: exit(1);
   default:
     printf("Wrong choice \n");
  }
 }
 return 0;
}
