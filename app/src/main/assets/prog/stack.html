Stack using Arary

#include ≺stdio.h≻
int stk[100], top=-1, n;

void show(){
  int i;
  for( i=0;i< top;i++){
    printf("%d->",stk[i]);
  }
   printf("null\n");
}

int push(){
   int item;
  if(top == n){
    printf("\nStack is full.\n");
    return 0;
  }
  if(top==-1){
    top = 0;
  }
  printf("Enter the item : ");
  scanf("%d",&item);
  stk[top]=item;
  top++;
  return 0;
}

void pop(){
  if(top==-1)
    printf("Stack is empty.");
  else{
    printf("% d is popped.\n",stk[top-1]);
    top--;
  }
}
int main(){
  int i,op;
  printf("Enter the size of the stack : ");
  scanf("%d",&n);
  do{
    printf("1: Push");
    printf(" 2: Pop");
    printf(" 3 : Print");
    printf(" 4 : Exit");
    printf("\nEnter your choice : ");
    scanf("%d",&op);
    switch(op){
      case 1: push();
       break;
      case 2: pop();
       break;
      case 3: show();
       break;
    }
  }while(op!=4);
return 0;
}
