void merge(int *a,int p,int q, int r){
  int i,j,k,*b;
  i=p;
  j=q+1;
  k=0;
  b=(int *)malloc(sizeof(int)*(r+1));
  while(i<=q && j<=r){
    if(a[i]<=a[j]){
      b[k++] = a[i++];
    }else{
      b[k++]=a[j++];
    }
  }
  while(i<=q){
    b[k++] = a[i++];
  }
  while(j<=r){
    b[k++]=a[j++];
  }
  k--;
  while(k>=0){
    a[p+k]=b[k];
    k--;
  }
}

void mergeSort(int *a,int p,int r){
  int q;
  if(p     q=(p+r)/2;
    mergeSort(a,p,q);
    mergeSort(a,q+1,r);
    merge(a,p,q,r);
  }
}
