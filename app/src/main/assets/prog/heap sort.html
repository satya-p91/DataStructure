void maxHeap(int *a,int i,int n){
  int j,element;
  j = 2*i;
  element = a[i];
  while(j<= n){
    if(j       j++;
    }
    if(element >= a[j])
      break;
    a[j/2] = a[j];
    j=2*j;
  }
  a[j/2] = element;
}

void buildHeap(int *a,int n){
   int i;
  for(i = n/2;i>=1;i--){
    maxHeap(a,i,n);
  }
}

  int *a,n,i,temp;

  for(i=1;i<=n;i++)
    scanf("%d",a+i);
  buildHeap(a,n);
  for(i=n;i>=2;i--){
    temp = a[1];
    a[1] = a[i];
    a[i] = temp;
    maxHeap(a,1,i-1);
  }
  printf("\nafter sorting:\n ");

    printf("%d ",a[i]);

}
