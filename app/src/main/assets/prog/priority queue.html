Proiority Queue using Array

#include≺stdio.h≻
#include≺process.h≻
#define MAX 10
int pri_que[MAX];
int front = -1, rear = -1;

int insert_by_priority(int data){
 if (rear >= MAX - 1){
  printf("\nQueue overflow");
  return 0;
 }
 if ((front == -1) && (rear == -1)){
  front++;
  rear++;
  pri_que[rear] = data;
  return 0;
 }
 else
  check(data);
 rear++;
 return 0;
}

int check(int data){
/*set priority based on higher value have more priority*/
 int i,j;
 for (i = 0; i <= rear; i++){
  if (data >= pri_que[i]){
   for (j = rear + 1; j > i; j--){
    pri_que[j] = pri_que[j - 1];
   }
   pri_que[i] = data;
   return 0;
  }
 }
 pri_que[i] = data;
 return 0;
}

int delete_by_priority(int data){
 int i;
 if ((front==-1) && (rear==-1)){
  printf("\nQueue is empty");
  return 0;
 }
 for (i = 0; i <= rear; i++){
  if (data == pri_que[i]){
   for (; i < rear; i++){
    pri_que[i] = pri_que[i + 1];
   }
   pri_que[i] = -99;
   rear--;
   if (rear == -1)
    front = -1;
   return 0;
  }
 }
 printf("\n%d not found in queue to delete", data);
 return 0;
}

int display(){
 if ((front == -1) && (rear == -1)){
  printf("\nQueue is empty");
  return 0;
 }
 for (; front <= rear; front++){
  printf(" %d ", pri_que[front]);
 }
 printf("\n");
 front = 0;
 return 0;
}

int main(){
 int n, ch;
 printf("1.Insert 2.Delete 3.Display 4.Exit\n");
 while (1){
  printf("Enter your choice : ");
  scanf("%d", &ch);
  switch (ch){
   case 1:
    printf("Enter value to be inserted : ");
    scanf("%d",&n);
    insert_by_priority(n);
    break;
   case 2:
    printf("Enter value to delete : ");
    scanf("%d",&n);
    delete_by_priority(n);
    break;
   case 3:
     display();
    break;
   case 4:
    exit(0);
   default:
    printf("\nWrong choice!");
  }
 }
 return 0;
}
